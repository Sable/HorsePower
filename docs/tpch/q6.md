## SQL 

```sql
  select
    sum(l_extendedprice * l_discount) as revenue
  from
    lineitem
  where
    l_shipdate >= date '1994-01-01'
    and l_shipdate < date '1994-01-01' + interval '1' year
    and l_discount between .06 - 0.01 and .06 + 0.01
    and l_quantity < 24; 
```

import Builtin;

    def main() : table {
        t0:f64        = check_cast(@column_value(`lineitem, `l_extendedprice), list<f64>);

        t2:d          = check_cast(@column_value(`lineitem, `l_shipdate     ), list<d>);
        t3:f64        = check_cast(@column_value(`lineitem, `l_quantity     ), list<f64>);

        t4:d          = 1994.01.01d
        t5:bool       = @ge(t3, t4);
        t6:d          = 1994.01.01d
        t7:d          = @datetime_add(t6, 1:i64, `year);
        t8:bool       = @lt(t2, t7);
        t9:f64        = @sub(0.06, 0.01);
        t10:f64       = @add(0.06, 0.01);
        t11:bool      = @between(t1, t9, t10);
        t12:bool      = @lt(t3, 24:i64);

        t13:bool      = @and(t5,t8);
        t14:bool      = @and(t13,t11);
        t15:bool      = @and(t14,t12);

        t16:f64       = @index(t0, t15);
        t17:f64       = @index(t1, t15);
        t18:f64       = @sum(t16, t17);

        t19:list<sym> = @tolist(`revenue);
        t20:list<f64> = @enlist(t18);
        t21:table     = @table(t19,t20);
        return t21;
    }
}
